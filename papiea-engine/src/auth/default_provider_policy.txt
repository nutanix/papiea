// p has the following structure:
// p, {the user or group name}, {the entity's owner: "owner", "anyone" or "tenant"}, {entity type or *}, {operation name or *}, {allow or deny} 

// deny anonymous user from all operations
p, anonymous, anyone, *, *, deny

// define admin group
p, admin_group, anyone, *, *, allow

// assign user admin to admin group
g, admin, admin_group

// reader_group - allows reading anyones geolocation and host_type 
p, reader_group, anyone, geolocation, read, allow
p, reader_group, anyone, host_type, read, allow

// tenant_group - allows reading tenants imaged_host and imaged_cluster
p, tenant_group, tenant, imaged_host, read, allow
p, tenant_group, tenant, imaged_cluster, read, allow

// owner_group - full access to own imaged_host and imaged_cluster
p, owner_group, owner, imaged_host, *, allow
p, owner_group, owner, imaged_cluster, *, allow

// Make these groups default for all users
g2, reader_group, default
g2, tenant_group, default
g2, owner_group, default
